The Toast component can be brought in for manual broadcast messages or
programmatic toasting.

### Manual broadcasting

```jsx example
function Example() {
    const [type, setType] = React.useState('success');

    return (
        <>
            <div className="mb-20">
                <h4 className="m-0">Bringing in the component manually</h4>
                <Toast type={type} text="Hello traveler!" canClose={false} />
            </div>
            <Toggle
                type="radio"
                onChange={(v) => setType(v.value)}
                defaultSelected={[{ label: 'Success', value: 'success' }]}
                options={[
                    { label: 'Success', value: 'success' },
                    { label: 'Error', value: 'error' },
                    { label: 'Warning', value: 'warning' },
                    { label: 'Info', value: 'info' },
                ]}
            />
        </>
    );
}
```

### Programmatically create a toast

```jsx example
function Example() {
    const { toast } = useToast();

    const [type, setType] = React.useState('success');

    return (
        <>
            <Toggle
                type="radio"
                onChange={(v) => setType(v.value)}
                defaultSelected={[{ label: 'Success', value: 'success' }]}
                options={[
                    { label: 'Success', value: 'success' },
                    { label: 'Error', value: 'error' },
                    { label: 'Warning', value: 'warning' },
                    { label: 'Info', value: 'info' },
                ]}
            />
            <h4 className="mt-10">Create toast</h4>
            <Button
                small
                onClick={() =>
                    toast("Hi! I'm an example default toast!", {
                        type,
                    })
                }
            >
                Default
            </Button>
            <Button
                small
                className="ml-10"
                onClick={() =>
                    toast("I won't go away, so you must close me!", {
                        type,
                        duration: 0,
                    })
                }
            >
                Disabled auto-removal
            </Button>
            <Button
                small
                className="ml-10"
                onClick={() =>
                    toast("You can't close me!", {
                        type,
                        canClose: false,
                    })
                }
            >
                Non closeable
            </Button>
        </>
    );
}

/* Root */
function App() {
    return (
        <ToastContainer>
            <Example />
        </ToastContainer>
    );
}
```

**Note** that you must wrap your App in the `<ToastContainer>` for toasts to
properly work.

**Note** that setting `duration: 0` will disable auto-dismissing.
